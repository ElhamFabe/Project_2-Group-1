const { Sequelize, DataTypes } = require('sequelize');
const sequelize = new Sequelize(/* ... */);
const queryInterface = sequelize.getQueryInterface();

module.exports = queryInterface.createTable('Search', {
            id: {
                type: DataTypes.INTEGER,
                primaryKey: true
            },
            title: {
                type: DataTypes.STRING,
                allowNull: false
            },
            category: {
                type: DataTypes.STRING,
                allowNull: false
            },
            category_id: {
                type: DataTypes.INTEGER,
                allowNull: false
            },
            gallery_picture: {
                type: DataTypes.STRING,
                allowNull: false
            }, 
            listing_type: {
                type: DataTypes.STRING,
                allowNull: false
            }, 
            current_price: {
                type: DataTypes.DECIMAL(10,2),
                allowNull: false
            },
            ending_time: {
                type: DataTypes.STRING,
                allowNull: false
            },
            watch_count: {
                type: DataTypes.STRING,
                allowNull: false
            }, 
            returns_accepted: {
                type: DataTypes.BOOLEAN,
                allowNull: false
            },
            shipping_type: {
                type: DataTypes.STRING,
                allowNull: false
            },
            url_link: {
                type: DataTypes.STRING, 
                allowNull: false
            },           
            top_rated_listing: {
                type: DataTypes.BOOLEAN,
                allowNull: false
            }
        });
